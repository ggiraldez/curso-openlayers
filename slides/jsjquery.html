Javascript

Funciones

function sumar(a, b) {
  var result;
  result = a + b;
  return result;
};
var c = sumar(10, 20);   // c == 30

Como las funciones son objetos de primera clase en Javascript, también podemos hacer 

var sumar = function(a, b) {
  return a + b;
}
var c = sumar(10, 20);   // c == 30

Javascript

Clausuras

Si una función usa una variable declarada por fuera de ella, por clausura la función captura una referencia a la variable que se utiliza cada vez que se invoca la función. 

var x = 2;
function sumar_x(a) {
  return a + x;
}
console.log(sumar_x(10));   // imprime 12
x = 5;
console.log(sumar_x(15));   // imprime 20

Javascript

Clausuras, un ejemplo más complicado

function crear_contador() {
  var i = 0;
  return function() {
    i = i + 1;
    return i;
  }
}
var c1 = crear_contador();
var c2 = crear_contador();
console.log(c1());   // 1
console.log(c1());   // 2
console.log(c1());   // 3
console.log(c2());   // 1
